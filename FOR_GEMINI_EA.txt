================================================================================
MT4 CODE TEMPLATE FOR GEMINI
================================================================================

//+------------------------------------------------------------------+
//| Expert Advisor Input Parameters                                  |
//+------------------------------------------------------------------+
input string API_KEY = "api_key_paste_your_key_here";  // Your FinDash OS API Key
input int SendIntervalSeconds = 60;                     // Send data every N seconds

//+------------------------------------------------------------------+
//| Global Variables                                                  |
//+------------------------------------------------------------------+
string API_URL = "https://oazfwlgnlrealwpyvqbu.supabase.co/functions/v1/portfolio-update";
datetime lastSendTime = 0;

//+------------------------------------------------------------------+
//| Expert Advisor Initialization                                     |
//+------------------------------------------------------------------+
int OnInit()
{
    // Validate API key format
    if(StringLen(API_KEY) < 40 || StringFind(API_KEY, "api_key_") != 0) {
        Alert("⚠️ Invalid API Key! Get your key from: Settings → Security in FinDash OS");
        return INIT_FAILED;
    }

    Print("✓ FinDash OS Portfolio Sync EA initialized");
    Print("✓ Account: ", AccountInfoInteger(ACCOUNT_LOGIN));
    Print("✓ API Key: ", StringSubstr(API_KEY, 0, 15), "...");

    return INIT_SUCCEEDED;
}

//+------------------------------------------------------------------+
//| Expert Advisor Tick Function                                      |
//+------------------------------------------------------------------+
void OnTick()
{
    // Check if it's time to send data
    datetime currentTime = TimeCurrent();

    if(currentTime - lastSendTime >= SendIntervalSeconds) {
        SendPortfolioData();
        lastSendTime = currentTime;
    }
}

//+------------------------------------------------------------------+
//| Send Portfolio Data to API                                        |
//+------------------------------------------------------------------+
void SendPortfolioData()
{
    // Build JSON request body
    string json = "{";
    json += "\"account_number\":" + IntegerToString(AccountInfoInteger(ACCOUNT_LOGIN)) + ",";
    json += "\"balance\":" + DoubleToString(AccountInfoDouble(ACCOUNT_BALANCE), 2) + ",";
    json += "\"equity\":" + DoubleToString(AccountInfoDouble(ACCOUNT_EQUITY), 2) + ",";
    json += "\"profit\":" + DoubleToString(AccountInfoDouble(ACCOUNT_PROFIT), 2) + ",";
    json += "\"server_time\":\"" + TimeToString(TimeCurrent(), TIME_DATE|TIME_SECONDS) + "\"";
    json += "}";

    // Build HTTP headers
    string headers = "Authorization: Bearer " + API_KEY + "\r\n";
    headers += "Content-Type: application/json\r\n";

    // Convert JSON string to char array
    char post[], result[];
    StringToCharArray(json, post, 0, StringLen(json));

    // Send HTTP POST request
    int httpCode = WebRequest(
        "POST",
        API_URL,
        headers,
        5000,  // 5 second timeout
        post,
        result,
        headers
    );

    // Handle response
    if(httpCode == 200) {
        Print("✓ Portfolio data sent successfully to FinDash OS");
    } else if(httpCode == 401) {
        Alert("⚠️ Unauthorized: Invalid API key. Check your key in Settings → Security");
    } else if(httpCode == 403) {
        Alert("⚠️ Forbidden: Your API key has been revoked. Generate a new one in Settings → Security");
    } else if(httpCode == -1) {
        Alert("⚠️ WebRequest failed! Make sure you added the URL to allowed URLs in Tools → Options → Expert Advisors");
    } else {
        Print("⚠️ Error sending data. HTTP code: ", httpCode);
    }
}
